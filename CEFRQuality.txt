    function createBarChart(canvasId, data, title, isSkillChart = false) {
        const ctx = document.getElementById(canvasId).getContext('2d');
        const chartData = {
            labels: cefrLevels,
            datasets: [{
                label: 'จำนวน (คน)', data: cefrLevels.map(level => data[level] || 0),
                backgroundColor: cefrLevels.map(level => cefrColors[level]),
                borderWidth: 1, borderRadius: 4,
            }]
        };
        const chartOptions = {
            responsive: true, maintainAspectRatio: false,
            plugins: {
                legend: { display: false },
                title: { display: true, text: title, font: { size: isSkillChart ? 14 : 16, weight: '500' }, padding: { bottom: 10 } }
            },
            scales: {
                y: { beginAtZero: true, ticks: { precision: 0, stepSize: 5, font: { size: 10 } }, grid: { color: '#e2e8f0' } },
                x: { grid: { display: false }, ticks: { font: { size: 12 } } }
            }
        };
        if (isSkillChart) {
             chartOptions.scales.y.ticks.stepSize = 4;
        }
        new Chart(ctx, { type: 'bar', data: chartData, options: chartOptions });
    }

            <tr class="hover:bg-slate-50">
                <td class="py-2 px-3 text-sm text-center text-slate-500">${index + 1}</td>
                <td class="py-2 px-3 text-sm text-slate-800 font-medium">${teacher.name}</td>

                <td class="py-2 px-3 text-center text-sm">${cefrBadge(teacher.r)}</td>
                <td class="py-2 px-3 text-center text-sm">${cefrBadge(teacher.l)}</td>
                <td class="py-2 px-3 text-center text-sm">${cefrBadge(teacher.s)}</td>
                <td class="py-2 px-3 text-center text-sm">${cefrBadge(teacher.w)}</td>
            </tr>
        `).join('');

        return `
            <div class="mb-8">
                <h3 class="text-xl font-semibold text-sky-700 mb-3">${schoolName}</h3>
                <div class="overflow-x-auto rounded-lg border border-slate-200">
                    <table class="min-w-full bg-white">
                        <thead class="bg-slate-100 text-slate-700">
                            <tr>
                                <th class="py-2 px-3 text-center text-sm font-semibold w-12">ลำดับ</th>
                                <th class="py-2 px-3 text-left text-sm font-semibold">ชื่อ-สกุล</th>
                                <th class="py-2 px-3 text-left text-sm font-semibold">ตำแหน่ง</th>
                                <th class="py-2 px-3 text-center text-sm font-semibold">Overall</th>
                                <th class="py-2 px-3 text-center text-sm font-semibold">Reading</th>
                                <th class="py-2 px-3 text-center text-sm font-semibold">Listening</th>
                                <th class="py-2 px-3 text-center text-sm font-semibold">Speaking</th>
                                <th class="py-2 px-3 text-center text-sm font-semibold">Writing</th>
                            </tr>
                        </thead>
                        <tbody class="divide-y divide-gray-200">${tableRows}</tbody>
                    </table>
                </div>
            </div>`;
    }

    // --- DOM MANIPULATION & CHART RENDERING ---
    document.addEventListener('DOMContentLoaded', () => {
        const schoolContainer = document.getElementById('infographic-container');
        const tablesContainer = document.getElementById('tables-container');
        const skillIcons = {reading: 'book-open', listening: 'ear', speaking: 'microphone', writing: 'pencil-line'};

        schoolData.forEach(school => {
            const schoolCard = document.createElement('div');
            schoolCard.className = 'card p-4 md:p-6';
            schoolCard.innerHTML = `
                <div class="border-b-2 border-slate-100 pb-4 mb-4">
                    <h2 class="text-xl md:text-2xl font-bold text-sky-700">${school.name}</h2>
                    <p class="text-slate-500">จำนวนผู้เข้าทดสอบ: ${school.participants} คน</p>
                </div>
                <div class="grid grid-cols-1 lg:grid-cols-2 gap-6 items-center">
                    <div>
                        <div class="chart-container"><canvas id="overall-${school.id}"></canvas></div>
                        <div class="mt-4 p-3 bg-slate-50 rounded-lg text-center">
                            <h3 class="font-semibold text-slate-700">ภาพรวม:</h3>
                            <p class="text-slate-600 text-sm">${school.summary}</p>
                        </div>
                    </div>
                    <div class="grid grid-cols-1 sm:grid-cols-2 gap-4">
                        ${Object.entries(school.skills).map(([key, value]) => `
                            <div class="p-2 border border-slate-200 rounded-lg">
                                 <div class="flex items-center gap-2 text-slate-600 mb-1">
                                    <i class="ph-fill ph-${skillIcons[key]}"></i><h4 class="font-medium text-sm">${value.name}</h4>
                                 </div>
                                 <div class="skill-chart-container"><canvas id="${key}-${school.id}"></canvas></div>
                            </div>
                        `).join('')}
                    </div>
                </div>`;
            schoolContainer.appendChild(schoolCard);

            if (allTeachersData[school.name]) {
                tablesContainer.innerHTML += createTable(school.name, allTeachersData[school.name]);
            }
        });

        createBarChart('overall-district', districtData.overall, 'ระดับความสามารถโดยรวม (Overall)');
        Object.entries(districtData.skills).forEach(([key, value]) => {
            createBarChart(`${key}-district`, value.data, '', true);
        });

        schoolData.forEach(school => {
            createBarChart(`overall-${school.id}`, school.overall, 'ระดับความสามารถโดยรวม (Overall)');
            Object.entries(school.skills).forEach(([key, value]) => {
                createBarChart(`${key}-${school.id}`, value.data, '', true);
            });
        });
    });
